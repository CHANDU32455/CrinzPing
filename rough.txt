import json
import boto3
from decimal import Decimal
from boto3.dynamodb.conditions import Key
from botocore.exceptions import ClientError
import time
import base64
from datetime import datetime

# --- AWS resources ---
dynamodb = boto3.resource("dynamodb")
s3 = boto3.client("s3")

CRINZ_TABLE = dynamodb.Table("CrinzMessages")
CONTENT_TABLE = dynamodb.Table("CrinzContent")
FOLLOWS_TABLE = dynamodb.Table("Follows")

BUCKET_CONTENT = "crinzcontent"
SIGNED_URL_EXPIRATION = 3600
FOLLOW_BATCH_SIZE = 10
PER_USER_CRINZ_LIMIT = 5
PER_USER_POST_LIMIT = 5
PER_USER_REEL_LIMIT = 5

# --- Helpers ---
def convert_decimal(obj):
    if isinstance(obj, list):
        return [convert_decimal(i) for i in obj]
    elif isinstance(obj, dict):
        return {k: convert_decimal(v) for k, v in obj.items()}
    elif isinstance(obj, Decimal):
        return int(obj) if obj % 1 == 0 else float(obj)
    return obj

def get_signed_url(bucket, key):
    if not key:
        return ""
    try:
        return s3.generate_presigned_url(
            'get_object', Params={'Bucket': bucket, 'Key': key}, ExpiresIn=SIGNED_URL_EXPIRATION
        )
    except ClientError:
        return ""

def decode_last_key(last_key):
    if not last_key:
        return None
    try:
        decoded = base64.b64decode(last_key).decode('utf-8')
        return json.loads(decoded)
    except:
        return None

def encode_last_key(last_key_dict):
    if not last_key_dict:
        return None
    try:
        json_str = json.dumps(last_key_dict)
        return base64.b64encode(json_str.encode('utf-8')).decode('utf-8')
    except:
        return None

# --- Fetch followees batch ---
def batch_get_followees(user_id, start_after=None, limit=FOLLOW_BATCH_SIZE):
    try:
        key_condition = Key("userId").eq(user_id) & Key("relationType#targetId").begins_with("following#")
        kwargs = {"KeyConditionExpression": key_condition, "Limit": limit}
        if start_after:
            kwargs["ExclusiveStartKey"] = start_after
        resp = FOLLOWS_TABLE.query(**kwargs)
        followees = [item["relationType#targetId"].split("#")[1] for item in resp.get("Items", [])]
        last_key = resp.get("LastEvaluatedKey")
        return followees, last_key
    except Exception as e:
        print(f"Error fetching followees: {e}")
        return [], None

# --- Fetch user content ---
def fetch_user_content(uid):
    reels, posts = [], []
    try:
        resp = CONTENT_TABLE.query(
            KeyConditionExpression=Key("pk").eq(f"USER#{uid}"),
            Limit=PER_USER_POST_LIMIT + PER_USER_REEL_LIMIT,
            ScanIndexForward=False
        )
        for item in resp.get("Items", []):
            files = [{"type": f.get("type"), "fileName": f.get("fileName"), "url": get_signed_url(BUCKET_CONTENT, f.get("s3Key"))} 
                     for f in item.get("files", []) if f.get("s3Key")]
            entry = {
                "id": item.get("postId"),
                "type": "reel" if item.get("type") == "crinzpostsreels" else "post",
                "userId": uid,
                "content": item.get("caption"),
                "tags": item.get("tags", []),
                "timestamp": int(item.get("timestamp", time.time() * 1000)),
                "likeCount": int(item.get("likes", 0)),
                "commentCount": int(item.get("comments", 0)),
                "files": files
            }
            if entry["type"] == "reel":
                reels.append(entry)
            else:
                posts.append(entry)
        reels = reels[:PER_USER_REEL_LIMIT]
        posts = posts[:PER_USER_POST_LIMIT]
    except Exception as e:
        print(f"Error fetching user content for {uid}: {e}")
    return posts + reels

# --- Fetch crinz messages ---
def fetch_crinz_messages(uid):
    items = []
    try:
        resp = CRINZ_TABLE.query(
            IndexName="userId-index",
            KeyConditionExpression=Key("userId").eq(uid),
            Limit=PER_USER_CRINZ_LIMIT,
            ScanIndexForward=False
        )
        for item in resp.get("Items", []):
            ts = item.get("timestamp")
            try:
                timestamp = int(ts) if ts else int(time.time() * 1000)
            except:
                timestamp = int(datetime.fromisoformat(ts).timestamp() * 1000) if ts else int(time.time() * 1000)
            crinz_item = {
                "id": item.get("crinzId", f"crinz-{uid}-{timestamp}"),
                "type": "crinz_message",
                "userId": uid,
                "userName": item.get("userName", "Unknown"),
                "content": item.get("message", ""),
                "tags": item.get("tags", []),
                "likeCount": int(item.get("likeCount", 0)),
                "commentCount": int(item.get("commentCount", 0)),
                "timestamp": timestamp,
                "files": []
            }
            if crinz_item["content"]:
                items.append(crinz_item)
    except Exception as e:
        print(f"Error fetching crinz messages for {uid}: {e}")
    return items

# --- Lambda handler ---
def lambda_handler(event, context):
    try:
        body = json.loads(event.get("body") or "{}")
        limit = int(body.get("limit", 15))
        last_key = body.get("lastKey")

        claims = event.get("requestContext", {}).get("authorizer", {}).get("jwt", {}).get("claims", {})
        user_id = claims.get("sub")
        if not user_id:
            return {"statusCode": 401, "body": json.dumps({"error": "Unauthorized"})}

        decoded_last_key = decode_last_key(last_key)
        processed_users = decoded_last_key.get("processedUsers", []) if decoded_last_key else []
        last_follow_key = decoded_last_key.get("followLastKey") if decoded_last_key else None
        seen_ids = set(decoded_last_key.get("seenIds", [])) if decoded_last_key else set()
        self_content_added = decoded_last_key.get("selfContentAdded", False) if decoded_last_key else False

        feed_items = []

        # --- Step 1: fetch followees batch ---
        followees_batch, new_follow_key = batch_get_followees(user_id, start_after=last_follow_key)
        for uid in followees_batch:
            if uid in processed_users:
                continue
            user_items = fetch_user_content(uid) + fetch_crinz_messages(uid)
            for item in user_items:
                if item["id"] not in seen_ids:
                    feed_items.append(item)
                    seen_ids.add(item["id"])
            processed_users.append(uid)
            if len(feed_items) >= limit:
                break

        # --- Step 2: fetch self content if feed not full and not already added ---
        if len(feed_items) < limit and not self_content_added:
            user_items = fetch_user_content(user_id) + fetch_crinz_messages(user_id)
            for item in user_items:
                if item["id"] not in seen_ids:
                    feed_items.append(item)
                    seen_ids.add(item["id"])
            self_content_added = True

        # --- Sort and trim ---
        feed_items.sort(key=lambda x: x.get("timestamp", 0), reverse=True)
        feed_items = feed_items[:limit]

        # --- Prepare next lastKey ---
        next_last_key = encode_last_key({
            "processedUsers": processed_users,
            "followLastKey": new_follow_key,
            "seenIds": list(seen_ids),
            "selfContentAdded": self_content_added
        }) if feed_items else None

        # --- Determine hasMore ---
        has_more = bool(new_follow_key) or (not self_content_added)

        return {
            "statusCode": 200,
            "body": json.dumps({
                "feedItems": convert_decimal(feed_items),
                "lastKey": next_last_key,
                "hasMore": has_more
            }),
            "headers": {
                "Content-Type": "application/json",
                "Access-Control-Allow-Origin": "*"
            }
        }

    except Exception as e:
        print(f"Error in lambda handler: {str(e)}")
        return {
            "statusCode": 500,
            "body": json.dumps({"error": str(e)}),
            "headers": {"Content-Type": "application/json", "Access-Control-Allow-Origin": "*"}
        }

import json
import boto3
import uuid
from datetime import datetime

dynamodb = boto3.resource("dynamodb")
likes_table = dynamodb.Table("CrinzLikes")
comments_table = dynamodb.Table("CrinzComments")
messages_table = dynamodb.Table("CrinzMessages")


def lambda_handler(event, context):
    try:
        # Handle CORS preflight
        if event.get("httpMethod") == "OPTIONS":
            return cors_response(200, {"success": True})

        body = json.loads(event.get("body", "{}"))
        actions = body.get("actions", [])
        user_id = body.get("userId")

        if not actions or not user_id:
            return cors_response(400, {"success": False, "error": "Missing actions or userId"})

        processed = []

        for action in actions:
            action_type = action.get("type")
            crinz_id = action.get("crinzId")
            payload = action.get("payload")
            timestamp = action.get("timestamp", str(int(datetime.now().timestamp() * 1000)))

            if not action_type or not crinz_id:
                processed.append({"type": action_type, "crinzId": crinz_id, "status": "invalid_action"})
                continue

            try:
                if action_type == "like":
                    result = handle_like(crinz_id, user_id, timestamp)
                elif action_type == "unlike":
                    result = handle_unlike(crinz_id, user_id)
                elif action_type == "add_comment":
                    comment_text = payload if isinstance(payload, str) else (payload.get("comment") if payload else "")
                    result = handle_add_comment(crinz_id, user_id, comment_text)
                elif action_type == "remove_comment":
                    comment_id = payload if isinstance(payload, str) else (payload.get("commentId") if payload else "")
                    result = handle_remove_comment(crinz_id, user_id, comment_id)
                else:
                    result = {"type": action_type, "crinzId": crinz_id, "status": "invalid_action_type"}
                processed.append(result)
            except Exception as e:
                processed.append({"type": action_type, "crinzId": crinz_id, "status": "error", "error": str(e)})

        return cors_response(200, {"success": True, "processed": processed})

    except Exception as e:
        return cors_response(500, {"success": False, "error": str(e)})


def cors_response(status_code, body):
    return {
        "statusCode": status_code,
        "headers": {
            "Access-Control-Allow-Origin": "*",
            "Access-Control-Allow-Headers": "Content-Type, Authorization",
            "Access-Control-Allow-Methods": "POST, OPTIONS"
        },
        "body": json.dumps(body)
    }


# Likes
def handle_like(crinz_id, user_id, timestamp):
    try:
        if likes_table.get_item(Key={"crinzId": crinz_id, "userId": user_id}).get("Item"):
            return {"type": "like", "crinzId": crinz_id, "status": "already_liked"}

        likes_table.put_item(Item={"crinzId": crinz_id, "userId": user_id, "timestamp": str(timestamp)})
        messages_table.update_item(
            Key={"crinzId": crinz_id},
            UpdateExpression="ADD likeCount :inc",
            ExpressionAttributeValues={":inc": 1},
            ConditionExpression="attribute_exists(crinzId)"
        )
        return {"type": "like", "crinzId": crinz_id, "status": "liked"}
    except messages_table.meta.client.exceptions.ConditionalCheckFailedException:
        likes_table.delete_item(Key={"crinzId": crinz_id, "userId": user_id})
        return {"type": "like", "crinzId": crinz_id, "status": "post_not_found"}


def handle_unlike(crinz_id, user_id):
    try:
        if not likes_table.get_item(Key={"crinzId": crinz_id, "userId": user_id}).get("Item"):
            return {"type": "unlike", "crinzId": crinz_id, "status": "not_liked"}

        likes_table.delete_item(Key={"crinzId": crinz_id, "userId": user_id})
        messages_table.update_item(
            Key={"crinzId": crinz_id},
            UpdateExpression="ADD likeCount :dec",
            ExpressionAttributeValues={":dec": -1},
            ConditionExpression="attribute_exists(crinzId)"
        )
        return {"type": "unlike", "crinzId": crinz_id, "status": "unliked"}
    except messages_table.meta.client.exceptions.ConditionalCheckFailedException:
        return {"type": "unlike", "crinzId": crinz_id, "status": "post_not_found"}


# Comments
def handle_add_comment(crinz_id, user_id, comment_text):
    if not comment_text or not isinstance(comment_text, str) or len(comment_text.strip()) == 0:
        return {"type": "add_comment", "crinzId": crinz_id, "status": "invalid_comment"}

    comment_id = str(uuid.uuid4())
    server_ts = str(int(datetime.now().timestamp() * 1000))
    comments_table.put_item(
        Item={
            "crinzId": crinz_id,
            "commentId": comment_id,
            "userId": user_id,
            "comment": comment_text.strip(),
            "timestamp": server_ts
        }
    )

    try:
        messages_table.update_item(
            Key={"crinzId": crinz_id},
            UpdateExpression="ADD commentCount :inc",
            ExpressionAttributeValues={":inc": 1},
            ConditionExpression="attribute_exists(crinzId)"
        )
    except messages_table.meta.client.exceptions.ConditionalCheckFailedException:
        comments_table.delete_item(Key={"crinzId": crinz_id, "commentId": comment_id})
        return {"type": "add_comment", "crinzId": crinz_id, "status": "post_not_found"}

    return {
        "type": "add_comment",
        "crinzId": crinz_id,
        "commentId": comment_id,
        "status": "comment_added",
        "payload": comment_text.strip()
    }


def handle_remove_comment(crinz_id, user_id, comment_id):
    if not comment_id:
        return {"type": "remove_comment", "crinzId": crinz_id, "status": "invalid_comment_id"}

    resp = comments_table.get_item(Key={"crinzId": crinz_id, "commentId": comment_id})
    item = resp.get("Item")
    if not item:
        return {"type": "remove_comment", "crinzId": crinz_id, "status": "comment_not_found"}

    if item["userId"] != user_id:
        return {"type": "remove_comment", "crinzId": crinz_id, "status": "not_owner"}

    comments_table.delete_item(Key={"crinzId": crinz_id, "commentId": comment_id})

    try:
        messages_table.update_item(
            Key={"crinzId": crinz_id},
            UpdateExpression="ADD commentCount :dec",
            ExpressionAttributeValues={":dec": -1},
            ConditionExpression="attribute_exists(crinzId)"
        )
    except messages_table.meta.client.exceptions.ConditionalCheckFailedException:
        pass

    return {
        "type": "remove_comment",
        "crinzId": crinz_id,
        "commentId": comment_id,
        "status": "comment_removed"
    }

import json
import boto3
from boto3.dynamodb.conditions import Key

dynamodb = boto3.resource("dynamodb")
comments_table = dynamodb.Table("CrinzComments")

def lambda_handler(event, context):
    try:
        body = {}
        if event.get("body"):
            body = json.loads(event["body"])

        crinz_id = body.get("crinzId")
        limit = int(body.get("limit", 15))
        last_key = body.get("lastKey")

        if not crinz_id:
            return {
                "statusCode": 400,
                "body": json.dumps({"error": "Missing crinzId"}),
                "headers": {
                    "Content-Type": "application/json",
                    "Access-Control-Allow-Origin": "*",
                },
            }

        query_params = {
            "KeyConditionExpression": Key("crinzId").eq(crinz_id),
            "Limit": limit,
        }
        if last_key:
            query_params["ExclusiveStartKey"] = last_key

        response = comments_table.query(**query_params)

        comments = response.get("Items", [])

        # Sort descending by timestamp so latest comments come first
        comments_sorted = sorted(comments, key=lambda c: c.get("timestamp", 0), reverse=True)

        return {
            "statusCode": 200,
            "body": json.dumps({
                "comments": comments_sorted,
                "lastKey": response.get("LastEvaluatedKey")
            }),
            "headers": {
                "Content-Type": "application/json",
                "Access-Control-Allow-Origin": "*",
            },
        }

    except Exception as e:
        return {
            "statusCode": 500,
            "body": json.dumps({"error": str(e)}),
            "headers": {
                "Content-Type": "application/json",
                "Access-Control-Allow-Origin": "*",
            },
        }
